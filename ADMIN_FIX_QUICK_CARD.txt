ON public.admin_users 
FOR SELECT 
TO authenticated

CREATE POLICY "super_admin_manage_admin_users" 
ON public.admin_users 
FOR ALL
TO authenticated 
USING (
  EXISTS (
    SELECT 1 FROM admin_users 
    WHERE user_id = auth.uid() 
    AND role = 'super_admin'
  )
);
